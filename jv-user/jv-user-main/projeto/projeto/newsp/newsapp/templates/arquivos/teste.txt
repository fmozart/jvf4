@login_required
def perfil(request):
    user = request.user
    try:
        perfil = Perfil.objects.get(user=user)
    except Perfil.DoesNotExist:
        return render(request, 'ci.html')

    comentarios = Comment.objects.filter(email=user.email).order_by('-created_at')  # Ordena por data, mais recentes primeiro

    return render(request, 'perfil.html', {'perfil': perfil, 'comentarios': comentarios})

def detail(request,id):
    news=News.objects.get(pk=id)
    if request.method=='POST':
        name=request.POST['name']
        email=request.POST['email']
        comment=request.POST['message'] 
        Comment.objects.create(
            news=news,
            name=name,
            email=email,
            comment=comment
        )
        messages.success(request,'comment submitted but in moderation mode.')
    category=Category.objects.get(id=news.category.id)
    rel_news=News.objects.filter(category=category).exclude(id=id)
    comments=Comment.objects.filter(news=news,status=True).order_by('-id')
